-- Two queries I have on the current athletes.php page.

1. Selects althete: first name + last name + age, team: name + age_group, COUNTS how many positions held.  Note: uses aggregate function

SELECT a.first_name, a.last_name, a.age, t.name, t.age_group, COUNT(p.type) AS Positions_Held FROM athletes a INNER JOIN teams t ON a.teamID = t.id INNER JOIN
athlete_position ap ON ap.athleteID = a.id INNER JOIN 
positions p ON p.id = ap.positionID 
GROUP BY a.first_name, a.last_name, a.age, t.name, t.age_group

2. All positions held by each athlete 

SELECT a.first_name, a.last_name, p.type FROM athletes a INNER JOIN athlete_position ap ON
ap.athleteID = a.id INNER JOIN positions p ON
p.id = ap.positionID
